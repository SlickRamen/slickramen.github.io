# Definition for singly-linked list. # class ListNode: # def __init__(self, val=0, next=None): # self.val = val # self.next = next class Solution: def modifiedList(self, nums: List[int], head: Optional[ListNode]) -> Optional[ListNode]: start = head prev = head num_set = set(nums) while start is not None: if start.val in num_set: if start == head: head = start.next else: prev.next = start.next else: prev = start start = start.next return head
