class Solution: def countValidSelections(self, nums: List[int]) -> int: n = len(nums) prefix = [0] * (n + 1) for i in range(n): prefix[i + 1] = prefix[i] + nums[i] result = 0 for i in range(n): if nums[i] == 0: if prefix[n] == 2 * prefix[i]: result += 2 elif abs(prefix[n] - 2 * prefix[i]) == 1: result += 1 return result
