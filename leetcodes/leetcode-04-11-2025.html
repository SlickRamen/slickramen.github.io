class Solution: def findXSum(self, nums: List[int], k: int, x: int) -> List[int]: n = len(nums) result = [] for i in range(n - k + 1): window = nums[i:i + k] freqs = Counter(window) freqs_sorted = sorted(freqs.items(), key=lambda freq: (-freq[1], -freq[0])) count_n = set(freq for freq, _ in freqs_sorted[:x]) sum_n = sum(val for val in window if val in count_n) result.append(sum_n) return result
